SELECT 
    h.hostName AS 'Nombre Host', 
    m.hmMAC AS 'Direccion MAC', 
    h.hostIP AS 'IP Reportada',
    i.imageName AS 'Imagen Asignada',
    o.osName AS 'Sistema Operativo'
FROM hosts h
INNER JOIN hostMAC m ON h.hostID = m.hmHostID AND m.hmPrimary = '1'
LEFT JOIN images i ON h.hostImage = i.imageID
LEFT JOIN supportedOS o ON i.imageOSID = o.osID;

Consultar Hosts por Grupo

SELECT 
    g.groupName AS 'Grupo',
    h.hostName AS 'Nombre Host', 
    m.hmMAC AS 'MAC'
FROM groups g
JOIN groupMembers gm ON g.groupID = gm.gmGroupID
JOIN hosts h ON gm.gmHostID = h.hostID
JOIN hostMAC m ON h.hostID = m.hmHostID AND m.hmPrimary = '1'
WHERE g.groupName = 'multicast_externo';

Consultar todos los Grupos y sus integrantes

SELECT 
    g.groupName, 
    h.hostName, 
    m.hmMAC
FROM groups g
LEFT JOIN groupMembers gm ON g.groupID = gm.gmGroupID
LEFT JOIN hosts h ON gm.gmHostID = h.hostID
LEFT JOIN hostMAC m ON h.hostID = m.hmHostID AND m.hmPrimary = '1'
ORDER BY g.groupName ASC;
