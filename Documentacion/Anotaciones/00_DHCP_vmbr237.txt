DHPC:

# 1. Crear el Pool
/ip pool add name=dhcp_pool237 ranges=172.18.30.100-172.18.30.250

# 2. Crear la red con múltiples DNS (Separados por coma)
/ip dhcp-server network
add address=172.18.30.0/24 gateway=172.18.30.1 dns-server=172.16.200.1,8.8.8.8 next-server=10.2.7.5

# 3. Crear el servidor DHCP
/ip dhcp-server
add name=vmbr237 interface=ether3 address-pool=dhcp_pool237 lease-time=10h disabled=no


/ip dhcp-server matcher
add name=match-legacy-237 code=93 value="0x0000" matching-type=exact server=vmbr237 address-pool=dhcp_pool237 option-set=set-legacy
